67// result.hpp - Rust-like Result<T, E> for C++17
68// SPDX-License-Identifier: MIT
69//
70// This header provides a modern, ergonomic Result type for C++17 inspired by
71// Rust. It supports combinators, helpers, and void specialization.
72//
73// All API is in the cpp_result namespace.
74//
75// --- API OVERVIEW ---
76//
77// Construction:
78// Result<T, E>::Ok(T)
79// Result<T, E>::Err(E)
80// Result<void, E>::Ok()
81// Result<void, E>::Err(E)
82// cpp_result::Ok<T, E>(T)
83// cpp_result::Err<T, E>(E)
84// cpp_result::Ok<E>()
85// cpp_result::Err<E>(E)
86//
87// Query:
88// is_ok(), is_err()
89//
90// Unwrap:
91// unwrap(), unwrap_err(), unwrap_or(value), unwrap_or_else(fn)
92// expect(msg), expect_err(msg)
93//
94// Combinators:
95// map(fn), map_err(fn), and_then(fn)
96// inspect(fn), inspect_err(fn)
97//
98// See the documentation below each function for usage examples.

148 ({ \
149 auto &&__res = (expr); \
150 using __res_type = std::decay_t<decltype(__res)>; \
151 if (__res.is_err()) \
152 return __res_type::Err(__res.unwrap_err()); \
