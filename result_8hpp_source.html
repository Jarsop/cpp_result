101// result.hpp - Rust-like Result<T, E> for C++17
102// SPDX-License-Identifier: MIT
103//
104// This header provides a modern, ergonomic Result type for C++17 inspired by
105// Rust. It supports combinators, helpers, and void specialization.
106//
107// All API is in the cpp_result namespace.
108//
109// --- API OVERVIEW ---
110//
111// Construction:
112// Result<T, E>::Ok(T)
113// Result<T, E>::Err(E)
114// Result<void, E>::Ok()
115// Result<void, E>::Err(E)
116// cpp_result::Ok<T, E>(T)
117// cpp_result::Err<T, E>(E)
118// cpp_result::Ok<E>()
119// cpp_result::Err<E>(E)
120//
121// Query:
122// is_ok(), is_err()
123//
124// Unwrap:
125// unwrap(), unwrap_err(), unwrap_or(value), unwrap_or_else(fn)
126// expect(msg), expect_err(msg)
127//
128// Combinators:
129// map(fn), map_err(fn), and_then(fn)
130// inspect(fn), inspect_err(fn)
131//
132// See the documentation below each function for usage examples.
